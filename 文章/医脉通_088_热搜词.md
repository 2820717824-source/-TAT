            function subscribe_class(class_id,flg = 0) {
                if (userid == 0) {
                    window.location.href = loginUrl;
                }
                if (!class_id) {
                    return false;
                }
                q = $("#yuyue_"+class_id);
                if(q.text() == '预约'){
                    type = 1;
                }else {
                    type = 2;
                }

                $.ajax({
                    url: '/new/branch/addClassSubscribe',
                    type: 'post',
                    dataType: 'json',
                    data: {
                        'userid': userid,
                        'class_id': class_id,
                        _token: $("#_token").val(),
                        'type':type
                    },
                    success: function (result) {
                        if (result.msg.success_msg != 'success') {
                            alert(result.msg.err_msg);
                            return false;
                        }else {
                            // $("#yuyue_"+class_id).attr('disabled',true);
                            if(flg == 1 ){
                                q.css("width", "100px");
                                q.css("height", "32px");
                            }

                            if(q.text() == '预约'){
                                q.attr('onclick','subscribe_class('+class_id+','+flg+')')
                                q.removeClass('blue-btn')
                                q.text('已预约')
                            }else {
                                q.attr('onclick','subscribe_class('+class_id+','+flg+')')
                                q.addClass('blue-btn')
                                q.text('预约')
                            }
                        }
                    }
                });
            }

            function getSubjectUrl(expert_id) {
                if (!expert_id) {
                    return false;
                }
                $.ajax({
                    url: '/new/getsubjecturl',
                    type: 'get',
                    dataType: 'json',
                    data: {
                        'expert_id': expert_id,
                        _token: $("#_token").val(),
                    },
                    success: function (result) {
                        if (result.code != 200) {
                            return false;
                        }else {
                            window.location.href = result.data.web_url;
                        }
                    }
                });
